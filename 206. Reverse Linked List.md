# using queue
```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */

class Solution {
public:
    ListNode* reverseList(ListNode* head) {
        ListNode* ln = nullptr;
        queue<int> qe;
        
        while (head != nullptr) {
            qe.push((*head).val);
            head = head->next;
        }
        while (!qe.empty()) {
            ln = new ListNode(qe.front(), ln);
            qe.pop();
        }
        return ln;
    }
};
```
